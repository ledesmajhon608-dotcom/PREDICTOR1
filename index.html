const KEY = "ea232d851cb4197391464e2c61df516b"; let db = {}; document.getElementById("liga").onchange = async function() { const liga = this.value; if(!liga) return; const loc = document.getElementById("local"); const vis = document.getElementById("visitante"); loc.innerHTML = vis.innerHTML = ""; // Intentamos temporadas 2024, 2025 o 2026 según disponibilidad const seasons = [2026, 2025, 2024]; let dataFound = false; for(let s of seasons) { if(dataFound) break; try { const res = await fetch(`https://v3.football.api-sports.io{liga}&season=${s}`, { headers: { "x-rapidapi-key": KEY, "x-rapidapi-host": "v3.football.api-sports.io" } }); const data = await res.json(); if(data.response && data.response.length > 0) { // CORRECCIÓN CLAVE: Acceder a la estructura correcta de standings const standingsGroups = data.response[0].league.standings; cargarEquipos(standingsGroups); dataFound = true; } } catch(e) { console.error("Error en temporada " + s, e); } } if(!dataFound) { loc.innerHTML = vis.innerHTML = ""; alert("No se encontraron datos para esta liga. Revisa tu API Key o conexión."); } }; function cargarEquipos(groups) { db = {}; const loc = document.getElementById("local"); const vis = document.getElementById("visitante"); loc.innerHTML = vis.innerHTML = ""; // Recorremos todos los grupos (por si la liga tiene varias tablas) groups.forEach(group => { group.forEach(t => { const name = t.team.name; const pj = t.all.played || 1; db[name] = { atk: t.all.goals.for / pj, def: t.all.goals.against / pj, form: (t.all.win * 3 + t.all.draw) / (pj * 3) }; loc.add(new Option(name, name)); vis.add(new Option(name, name)); }); }); } function poisson(lambda, k) { let fact = 1; for(let i=1; i<=k; i++) fact *= i; return (Math.exp(-lambda) * Math.pow(lambda, k)) / fact; } function calcular() { const lName = document.getElementById("local").value; const vName = document.getElementById("visitante").value; if(!db[lName] || lName === vName) return alert("Selecciona equipos válidos"); const L = db[lName]; const V = db[vName]; // xG Ajustado let expL = (L.atk * V.def * 1.25) + (L.form * 0.3); let expV = (V.atk * L.def * 0.85) + (V.form * 0.2); let pL = 0, pE = 0, pV = 0; for(let i=0; i<6; i++) { for(let j=0; j<6; j++) { const prob = poisson(expL, i) * poisson(expV, j); if(i > j) pL += prob; else if(i === j) pE += prob; else pV += prob; } } const total = pL + pE + pV; const resL = (pL/total*100).toFixed(1); const resE = (pE/total*100).toFixed(1); const resV = (pV/total*100).toFixed(1); document.getElementById("resultado").innerHTML = `
${lName} vs ${vName}
Local: ${resL}%
Empate: ${resE}%
Visitante: ${resV}%
Pick: ${parseFloat(resL) > 45 ? 'Gana Local' : (parseFloat(resV) > 45 ? 'Gana Visitante' : 'Doble Oportunidad')}
`; }
