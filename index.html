FUNCIÓN PARA OBTENER DATOS REALES async function cargarDatosLiga(ligaId) { const url = `https://v3.football.api-sports.io/standings?league=${ligaId}&season=2024`; // Temporada actual try { const response = await fetch(url, { "method": "GET", "headers": { "x-rapidapi-host": "v3.football.api-sports.io", "x-rapidapi-key": API_KEY } }); const data = await response.json(); const standings = data.response[0].league.standings[0]; // Limpiar selects y stats localSelect.innerHTML = ""; visitanteSelect.innerHTML = ""; statsDinamicas = {}; standings.forEach(item => { const team = item.team.name; const pj = item.all.played; // Calculamos stats en base a rendimiento real statsDinamicas[team] = { atk: (item.all.goals.for / pj).toFixed(2), def: (item.all.goals.against / pj).toFixed(2) }; localSelect.add(new Option(team, team)); visitanteSelect.add(new Option(team, team)); }); console.log("Datos actualizados para:", data.response[0].league.name); } catch (error) { alert("Error al conectar con la API. Verifica tu API Key."); } } // Escuchar cuando el usuario cambia de liga ligaSelect.addEventListener("change", function() { if (this.value) { cargarDatosLiga(ligasIDs[this.value]); } }); function calcular() { const loc = localSelect.value; const vis = visitanteSelect.value; if (!loc || !vis || loc === vis) return; // Usamos los datos recién traídos de la API const sL = statsDinamicas[loc]; const sV = statsDinamicas[vis]; let expL = (sL.atk * sV.def * 1.15); let expV = (sV.atk * sL.def); let total = expL + expV + 1.2; let pL = ((expL / total) * 100).toFixed(1); let pV = ((expV / total) * 100).toFixed(1); let pE = (100 - pL - pV).toFixed(1); let gTotales = parseFloat(expL) + parseFloat(expV); let p25 = Math.min(90, (gTotales * 25)).toFixed(1); // (El resto de la función de renderizado HTML es igual a la que ya tenías) renderizarResultado(loc, vis, pL, pE, pV, gTotales, p25); } function renderizarResultado(loc, vis, pL, pE, pV, gTotales, p25) { document.getElementById("contenedor-resultado").innerHTML = `
LOCAL ${loc}
VS
VISITA ${vis}
Victoria Local${pL}%
Empate${pE}%
Victoria Visita${pV}%
Ambos Marcan
${(pL>30&&pV>30)?'SÍ':'NO'}
Goles +2.5
${p25}%
Córners Est.
${Math.floor(7+(gTotales*1.5))}
Goles +1.5
${Math.min(99, (gTotales*40)).toFixed(1)}%
RECOMENDACIÓN
${pL > 45 ? 'Gana ' + loc : pV > 45 ? 'Gana ' + vis : 'Doble Oportunidad / Empate'}
`; setTimeout(() => { document.getElementById("b1").style.width = pL + "%"; document.getElementById("b2").style.width = pE + "%"; document.getElementById("b3").style.width = pV + "%"; }, 100); }
